# Relaissteuerung mit ioBroker und dem Raspberry Pi

### Voraussetzungen

##### Installierte Adapter:
* rpi2
* javascript
* vis

### Datenpunkte

* javascript.0
    * Ventilsteuerung
        * Relais(n)
            * ##### dauer 
            >Dauer, die ein Ventil geöffnet sein soll in Minuten

            * ##### stunde
            > Stunde, zu der das Ventil geöffnet werden soll (0 - 23)

            * ##### minute
            > Minute, zu der das Ventil geöffnet werden soll (0 - 59)

            * ##### wartung
            > Ist dieser Wert auf true, wird das Relais nicht aktiviert (Standard: true)

            * ##### status
            > Sollzustand des Ventils

            * ##### gpio
            > Der Pin, an dem das Relais angeschlossen ist (BCM-Nummerierung)
         * ##### Refresh
         > Bei Änderung auf true wird signalisiert, dass der Zeitplan neu erstellt werden soll, was zum Beispiel beim Speichern im vis der Fall ist


### GPIO
In den Einstellungen des Adapters rpi2.0 müssen die entsprechenden GPIO aktiviert und auf OUT gesetzt werden. 

### Skripte

#### Erstellen der Datenpunkte
Zum Erstellen der Datenpunkte kann das folgende Skript genutzt werden. Dazu legt man im ioBroker unter "Skripte" ein neues Skript an, nennt es beispielsweise "Datenpunkte_erstellen" und fügt folgenden Code ein. Das Skript muss nur ein Mal ausgeführt und anschließend wieder gestoppt werden. Möchte man die Datenpunkte erneut erstellen, sollte man vorher den Punkt "javascript.0.Ventilsteuerung" löschen. 
```javascript 
let count = 8;

for (var i = 1; i <= count; i++) {
    createState("Ventilsteuerung.Relais" + i + ".stunde", 0, {read: true, write: true, name: "Stunde", type: "number", role: "value"});
    createState("Ventilsteuerung.Relais" + i + ".minute", 0, {read: true, write: true, name: "Minute", type: "number", role: "value"});
    createState("Ventilsteuerung.Relais" + i + ".dauer", 60, {read: true, write: true, name: "Dauer", type: "number", role: "value"});
    createState("Ventilsteuerung.Relais" + i + ".wartung", true, {read: true, write: true, name: "Wartung", type: "boolean"});
    createState("Ventilsteuerung.Relais" + i + ".status", false, {read: true, write: true, name: "Status", type: "boolean"});
    createState("Ventilsteuerung.Relais" + i + ".gpio", 0, {read: true, write: true, name: "GPIO", type: "number", role: "value"});
}

createState("javascript.0.Ventilsteuerung.Refresh", false, {read: true, write: true, name: "Refresh", type: "boolean"});

log("Datenpunkte erstellt.", "info");
```

#### Relaissteuerung
Das anzulegende Skript "Relaissteuerung" beinhaltet die Zeitplanung und setzt den Datenpunkt "status" eines Relais. Wird eine Änderung festgestellt, wird der entsprechende Pin geschaltet.
```javascript
let scheduleList = [];

addRelais("javascript.0.Ventilsteuerung.Relais1");
addRelais("javascript.0.Ventilsteuerung.Relais2");
addRelais("javascript.0.Ventilsteuerung.Relais3");
addRelais("javascript.0.Ventilsteuerung.Relais4");
addRelais("javascript.0.Ventilsteuerung.Relais5");
addRelais("javascript.0.Ventilsteuerung.Relais6");
addRelais("javascript.0.Ventilsteuerung.Relais7");
addRelais("javascript.0.Ventilsteuerung.Relais8");
refreshSchedules();


function addRelais(id) {
    scheduleList.push({"id": id, "schedule": null});

    let gpio = getState(id + ".gpio").val;
    on(id + ".status", function(obj) {
        if (!obj.state.ack && obj.state.val) { 
            if (gpio > 0) {
                setState("rpi2.0.gpio." + gpio + ".state", obj.state);        
            }            
        }
    });
}

function refreshSchedules() {
    clearScheduleList();    
    scheduleList.forEach(function(relais){
        relais.schedule = getSchedule(relais);
    });
}

function getSchedule(relais) {
    let minute = getState(relais.id + ".minute").val;
    let stunde = getState(relais.id + ".stunde").val;
    return schedule("0 " + minute + " " + stunde + " * * *", activateRelais);
};

function clearScheduleList() {
    scheduleList.forEach(function(relais){
        clearSchedule(relais.schedule);
    });
}

function activateRelais(relais) {
    let duration = getState(relais.id + ".dauer").val * 1000 * 60;
    let gpio = getState(relais.id + ".gpio").val;
    
    setState(relais.id + ".status", true);    
    setStateDelayed(relais.id + ".status", false, duration);
}

on('javascript.0.Ventilsteuerung.Refresh', function (obj) {
    if (!obj.state.ack && obj.state.val) {                
        refreshSchedules();
        setState('javascript.0.Ventilsteuerung.Refresh', false );        
    }
});
```
